## Phase 2：新增 `VectorBackedRAGService`（~45 分钟）

### 2.1 创建新文件

**文件**：`backend/app/services/rag/chroma_retriever.py`

```python
from __future__ import annotations

from app.services.rag.base import BaseRAGService, RetrievedDocument
from app.services.rag.embedder import BaseEmbedder
from app.services.rag.query_rewriter import QueryRewriter
from app.services.rag.reranker import Reranker
from app.services.rag.vector_store import BaseVectorStore


class VectorBackedRAGService(BaseRAGService):
    def __init__(
        self,
        vector_store: BaseVectorStore,
        embedder: BaseEmbedder,
        query_rewriter: QueryRewriter,
        reranker: Reranker,
    ) -> None:
        self.vector_store = vector_store
        self.embedder = embedder
        self.query_rewriter = query_rewriter
        self.reranker = reranker

    async def retrieve(
        self, query: str, normalized_query: str | None = None, top_k: int = 5
    ) -> list[RetrievedDocument]:
        normalized = normalized_query or self.query_rewriter.rewrite(query)
        query_embedding = self.embedder.embed(normalized)
        candidates = self.vector_store.query(query_embedding, top_k=20)

        documents: list[RetrievedDocument] = []
        for result in candidates:
            meta = result.metadata
            documents.append(
                RetrievedDocument(
                    title=meta.get("title", ""),
                    content=result.document,
                    snippet=result.document[:200],
                    source=meta.get("source", ""),
                    score=result.score,
                )
            )

        ranked = self.reranker.rerank(query, documents, top_k=top_k)
        return [item.document for item in ranked]
```

## Phase 3：新增 `CrossEncoderReranker`（~30 分钟）

### 3.1 创建新文件

**文件**：`backend/app/services/rag/cross_encoder_reranker.py`

```python
from __future__ import annotations

from app.core.exceptions import AppError
from app.services.rag.reranker import Reranker, RerankResult


class CrossEncoderReranker(Reranker):
    def __init__(self, model_name: str = "BAAI/bge-reranker-base"):
        self.model_name = model_name
        try:
            from sentence_transformers import CrossEncoder
        except ImportError as exc:
            raise AppError(
                "sentence-transformers is not installed.",
                error_code="rag_dependency_missing",
                status_code=503,
            ) from exc
        self._model = CrossEncoder(model_name)

    def rerank(self, query, documents, top_k=5):
        if not documents:
            return []
        pairs = [(query, doc.content) for doc in documents]
        scores = self._model.predict(pairs, show_progress_bar=False)
        ranked = sorted(
            zip(documents, scores), key=lambda x: x[1], reverse=True
        )
        return [
            RerankResult(document=doc, score=float(score))
            for doc, score in ranked[:top_k]
        ]


class BM25Reranker(Reranker):
    """Fallback reranker when CrossEncoder model is unavailable."""

    def __init__(self):
        import jieba
        self._tokenize = jieba.lcut
        from rank_bm25 import BM25Okapi as _BM25
        self._BM25 = _BM25

    def rerank(self, query, documents, top_k=5):
        if not documents:
            return []
        corpus = [self._tokenize(doc.content) for doc in documents]
        bm25 = self._BM25(corpus)
        scores = bm25.get_scores(self._tokenize(query))
        ranked = sorted(
            zip(documents, scores), key=lambda x: x[1], reverse=True
        )
        return [
            RerankResult(document=doc, score=float(score))
            for doc, score in ranked[:top_k]
        ]
```

```python
import json
import httpx

async def stream_generate(self, query: str, documents: list[RetrievedDocument]):
    if not documents:
        yield "目前没有足够的景区证据可以组织可靠回答。"
        return

    context = "\n---\n".join(
        f"[来源：{doc.source}] {doc.content}" for doc in documents
    )
    messages = [
        {"role": "system", "content": self.settings.llm_system_prompt},
        {"role": "user", "content": f"参考资料：\n{context}\n\n问题：{query}"},
    ]

    async with httpx.AsyncClient(timeout=60) as client:
        async with client.stream(
            "POST",
            f"{self.settings.llm_base_url}/chat/completions",
            headers={
                "Authorization": f"Bearer {self.settings.llm_api_key}",
                "Content-Type": "application/json",
            },
            json={
                "model": self.settings.llm_model,
                "messages": messages,
                "stream": True,
                "temperature": 0.3,
                "max_tokens": 1024,
            },
        ) as response:
            async for line in response.aiter_lines():
                if not line.startswith("data: "):
                    continue
                if line.strip() == "data: [DONE]":
                    break
                try:
                    delta = json.loads(line[6:])["choices"][0]["delta"]
                    content = delta.get("content", "")
                    if content:
                        yield content
                except (json.JSONDecodeError, KeyError, IndexError):
                    continue
```

```python
# eval/metrics/accuracy.py
from __future__ import annotations
import json
from pathlib import Path

def evaluate(testset_path: Path, answer_fn) -> dict:
    """answer_fn(query) -> str"""
    cases = json.loads(testset_path.read_text(encoding="utf-8"))
    passed = 0
    failures = []
    for case in cases:
        answer = answer_fn(case["query"])
        must_contain = case.get("must_contain", [])
        if all(token in answer for token in must_contain):
            passed += 1
        else:
            failures.append({
                "id": case["id"],
                "query": case["query"],
                "expected_contains": must_contain,
                "actual_answer": answer[:200],
            })
    return {
        "total": len(cases),
        "passed": passed,
        "accuracy": passed / len(cases) if cases else 0,
        "failures": failures,
    }
```

```python
from __future__ import annotations
from dataclasses import dataclass
from snownlp import SnowNLP

@dataclass(slots=True)
class SentimentResult:
    score: float
    label: str  # "positive" / "neutral" / "negative"

class SentimentAnalyzer:
    def analyze(self, text: str) -> SentimentResult:
        s = SnowNLP(text)
        score = s.sentiments
        if score > 0.7:
            label = "positive"
        elif score < 0.3:
            label = "negative"
        else:
            label = "neutral"
        return SentimentResult(score=score, label=label)
```

```python
@router.get("/sentiment-trend")
async def sentiment_trend(
    days: int = 7,
    session: Session = Depends(get_db_session),
):
    from collections import defaultdict
    logs = ChatLogRepository(session).list_recent(days=days)
    stats = defaultdict(lambda: {"positive": 0, "neutral": 0, "negative": 0})
    for log in logs:
        day = log.created_at.strftime("%Y-%m-%d")
        label = log.sentiment_label or "neutral"
        stats[day][label] += 1
    return {"trend": dict(stats), "days": days}
```
